tree("Root")
	parallel
		repeat mute tree ("Patrulhar")
		repeat mute tree ("Atacar")
		repeat mute tree ("Fugir")


tree ("Patrulhar")
	fallback
		while not IsHealthLessThan(60.0)
			fallback
				while ninguem
					sequence
						PickRandomDestination
						Parar

tree ("Atacar")
	fallback
		while not IsHealthLessThan(60.0)
			fallback
				while not ninguem
					sequence
						Atacar

tree("Fugir")
	fallback
		while IsHealthLessThan(60.0)
			fallback
				while not ninguem
					sequence
						TakeCover
						Parar
				fallback
					while ninguem
						sequence
							PickDestination(0.0,0.0)
							Parar
